User Model  - id (PK, UUID or INT AUTO_INCREMENT)
            - email (VARCHAR UNIQUE, indexed)
            - password_hash (VARCHAR)
            - first_name (VARCHAR)
            - last_name (VARCHAR)
            - phone (VARCHAR)
            - role (ENUM: 'customer', 'admin')
            - created_at (TIMESTAMP DEFAULT NOW)
            - updated_at (TIMESTAMP DEFAULT NOW ON UPDATE NOW)
Adresses Model  - id (PK)
                - user_id (FK to Users.id)
                - type (ENUM: 'shipping', 'billing')
                - street (TEXT)
                - city (VARCHAR)
                - state (VARCHAR)
                - postal_code (VARCHAR)
                - country (VARCHAR)
                - is_default (BOOLEAN DEFAULT FALSE)

Categories Model    - id (PK)
                    - name (VARCHAR UNIQUE)
                    - slug (VARCHAR UNIQUE, for URLs)
                    - parent_id (FK to Categories.id, NULL for top-level)
                    - description (TEXT)
                    - image_url (VARCHAR)
Products Model      - id (PK)
                    - name (VARCHAR, indexed)
                    - slug (VARCHAR UNIQUE, for URLs)
                    - description (TEXT)
                    - price (DECIMAL(10,2))
                    - stock_quantity (INT DEFAULT 0)
                    - sku (VARCHAR UNIQUE)
                    - image_url (VARCHAR)
                    - category_id (FK to Categories.id)
                    - is_active (BOOLEAN DEFAULT TRUE)
                    - created_at (TIMESTAMP DEFAULT NOW)
Carts Model         - id (PK)
                    - user_id (FK to Users.id, NULL for guests)
                    - session_id (VARCHAR, for guests)
                    - created_at (TIMESTAMP DEFAULT NOW)
                    - updated_at (TIMESTAMP DEFAULT NOW ON UPDATE NOW)
Cart Items          - id (PK)
                    - cart_id (FK to Carts.id)
                    - product_id (FK to Products.id)
                    - quantity (INT DEFAULT 1)
                    - added_at (TIMESTAMP DEFAULT NOW)

Orders Models       - id (PK)
                    - user_id (FK to Users.id)
                    - shipping_address_id (FK to Addresses.id)
                    - billing_address_id (FK to Addresses.id)
                    - order_number (VARCHAR UNIQUE, e.g., 'ORD-2025-001')
                    - total_amount (DECIMAL(10,2))
                    - status (ENUM: 'pending', 'paid', 'shipped', 'delivered', 'cancelled')
                    - order_date (TIMESTAMP DEFAULT NOW)
                    - shipped_at (TIMESTAMP NULL)
OrderItems Model    - id (PK)
                    - order_id (FK to Orders.id)
                    - product_id (FK to Products.id)
                    - quantity (INT)
                    - unit_price (DECIMAL(10,2))
Payments model      - id (PK)
                    - order_id (FK to Orders.id)
                    - payment_method (ENUM: 'credit_card', 'paypal', 'bank_transfer')
                    - amount (DECIMAL(10,2))
                    - status (ENUM: 'pending', 'completed', 'failed')
                    - transaction_id (VARCHAR)
                    - paid_at (TIMESTAMP NULL)
Reviews             - id (PK)
                    - user_id (FK to Users.id)
                    - product_id (FK to Products.id)
                    - rating (INT 1-5)
                    - comment (TEXT)
                    - created_at (TIMESTAMP DEFAULT NOW)
                    - is_approved (BOOLEAN DEFAULT FALSE)
